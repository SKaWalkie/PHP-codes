🔴 Q12 – Property + Owner
CREATE TABLE Owner(
  oname VARCHAR(50) PRIMARY KEY,
  address VARCHAR(100),
  phone VARCHAR(15)
);

CREATE TABLE Property(
  pno INT,
  description VARCHAR(100),
  area DOUBLE PRECISION,
  oname VARCHAR(50),
  FOREIGN KEY(oname) REFERENCES Owner(oname)
);
